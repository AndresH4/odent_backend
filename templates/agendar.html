from flask import Blueprint, request, jsonify from db import get_db_connection citas_bp = Blueprint('citas_bp', __name__) # ───────────────────────────────────────────── # UTILIDADES # ───────────────────────────────────────────── def _rows_to_list(cursor): return [dict(row) for row in cursor.fetchall()] def _json_ok(data, code=200): return jsonify(data), code def _json_error(mensaje, code=400): return jsonify({"ok": False, "error": mensaje}), code # ───────────────────────────────────────────── # ESPECIALISTAS # ───────────────────────────────────────────── @citas_bp.route('/api/especialistas', methods=['GET']) def get_especialistas(): con = None try: con = get_db_connection() cur = con.cursor() cur.execute(""" SELECT e.Especialista_ID, u.Nombres || ' ' || u.Apellidos AS NombreCompleto, GROUP_CONCAT(esp.Nombre_Especialidad, ', ') AS Especialidades, e.Tarjeta_Profesional FROM especialistas e JOIN usuarios u ON u.Usuario_ID = e.Usuario_ID LEFT JOIN especialista_especialidad ee ON ee.Especialista_ID = e.Especialista_ID LEFT JOIN especialidad esp ON esp.Especialidad_ID = ee.Especialidad_ID GROUP BY e.Especialista_ID """) return _json_ok(_rows_to_list(cur)) except Exception as exc: return _json_error(str(exc), 500) finally: if con: con.close() # ───────────────────────────────────────────── # AGENDA # ───────────────────────────────────────────── @citas_bp.route('/api/agenda', methods=['GET']) def get_agenda(): esp_id = request.args.get('especialista_id') fecha = request.args.get('fecha') con = None try: con = get_db_connection() cur = con.cursor() sql = """ SELECT a.Agenda_ID, a.Especialista_ID, a.Fecha, a.Hora_Inicio, a.Hora_Fin, ea.Nombre_Estado AS EstadoAgenda, esp.Nombre_Especialidad, u.Nombres || ' ' || u.Apellidos AS NombreEspecialista FROM agenda a JOIN estado_agenda ea ON ea.Estado_ID = a.Estado_ID JOIN especialistas e ON e.Especialista_ID = a.Especialista_ID JOIN usuarios u ON u.Usuario_ID = e.Usuario_ID LEFT JOIN especialista_especialidad ee ON ee.Especialista_ID = e.Especialista_ID LEFT JOIN especialidad esp ON esp.Especialidad_ID = ee.Especialidad_ID WHERE ea.Nombre_Estado = 'Disponible' """ params = [] if esp_id: sql += " AND a.Especialista_ID = ?" params.append(esp_id) if fecha: sql += " AND a.Fecha = ?" params.append(fecha) sql += " ORDER BY a.Fecha, a.Hora_Inicio" cur.execute(sql, params) return _json_ok(_rows_to_list(cur)) except Exception as exc: return _json_error(str(exc), 500) finally: if con: con.close() # ───────────────────────────────────────────── # CITAS (LISTAR) # ───────────────────────────────────────────── @citas_bp.route('/api/citas', methods=['GET']) def get_citas(): paciente_id = request.args.get('paciente_id') con = None try: con = get_db_connection() cur = con.cursor() sql = """ SELECT c.Cita_ID, c.Motivo_Consulta, a.Fecha, a.Hora_Inicio, a.Hora_Fin FROM cita c JOIN agenda a ON a.Agenda_ID = c.Agenda_ID WHERE 1=1 """ params = [] if paciente_id: sql += " AND c.Paciente_ID = ?" params.append(paciente_id) cur.execute(sql, params) return _json_ok(_rows_to_list(cur)) except Exception as exc: return _json_error(str(exc), 500) finally: if con: con.close() # ───────────────────────────────────────────── # CREAR CITA # ───────────────────────────────────────────── @citas_bp.route('/api/citas', methods=['POST']) def crear_cita(): datos = request.get_json(silent=True) or {} paciente_id = datos.get('Paciente_ID') agenda_id = datos.get('Agenda_ID') motivo = (datos.get('Motivo_Consulta') or '').strip() if not all([paciente_id, agenda_id, motivo]): return _json_error("Faltan datos obligatorios") con = None try: con = get_db_connection() cur = con.cursor() # validar agenda cur.execute(""" SELECT Estado_ID FROM agenda WHERE Agenda_ID = ? """, (agenda_id,)) agenda = cur.fetchone() if not agenda: return _json_error("Agenda no existe", 404) # validar disponibilidad cur.execute(""" SELECT Nombre_Estado FROM estado_agenda WHERE Estado_ID = ? """, (agenda["Estado_ID"],)) estado = cur.fetchone() if not estado or estado["Nombre_Estado"] != "Disponible": return _json_error("Horario no disponible") # insertar cita cur.execute(""" INSERT INTO cita (Paciente_ID, Agenda_ID, Motivo_Consulta) VALUES (?, ?, ?) """, (paciente_id, agenda_id, motivo)) # marcar ocupada cur.execute(""" UPDATE agenda SET Estado_ID = ( SELECT Estado_ID FROM estado_agenda WHERE Nombre_Estado = 'Ocupado' ) WHERE Agenda_ID = ? """, (agenda_id,)) con.commit() return _json_ok({"ok": True}) except Exception as exc: if con: con.rollback() return _json_error(str(exc), 500) finally: if con: con.close() # ───────────────────────────────────────────── # MULTA ACTIVA # ───────────────────────────────────────────── @citas_bp.route('/api/paciente//multa-activa', methods=['GET']) def multa_activa(paciente_id): con = None try: con = get_db_connection() cur = con.cursor() cur.execute(""" SELECT m.Multa_ID FROM multa m JOIN cita c ON m.Cita_ID = c.Cita_ID JOIN estado_multa em ON m.Estado_Multa_ID = em.Estado_Multa_ID WHERE c.Paciente_ID = ? AND em.Nombre_Estado = 'Pendiente' LIMIT 1 """, (paciente_id,)) row = cur.fetchone() return _json_ok({"tiene_multa": row is not None}) except Exception as exc: return _json_error(str(exc), 500) finally: if con: con.close()
